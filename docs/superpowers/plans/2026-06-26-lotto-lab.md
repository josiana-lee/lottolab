// src/lib/combo.ts
import { LOW_NUMBER_THRESHOLD, MIN_SUM, MAX_SUM } from './constants'

export function createComboKey(numbers: number[]): string {
  return [...numbers].sort((a, b) => a - b).join('-')
}

export function isValidCombo(numbers: number[], pastComboSet: Set<string>): boolean {
  const sorted = [...numbers].sort((a, b) => a - b)
  const key = sorted.join('-')

  if (pastComboSet.has(key)) return false

  const sum = sorted.reduce((a, b) => a + b, 0)
  if (sum < MIN_SUM || sum > MAX_SUM) return false

  const oddCount = sorted.filter(n => n % 2 === 1).length
  if (oddCount === 0 || oddCount === 6) return false

  const lowCount = sorted.filter(n => n <= LOW_NUMBER_THRESHOLD).length
  if (lowCount === 0 || lowCount === 6) return false

  // 연속번호 4개 이상 제외
  let maxConsec = 1, cur = 1
  for (let i = 1; i < sorted.length; i++) {
    cur = sorted[i] === sorted[i - 1] + 1 ? cur + 1 : 1
    if (cur > maxConsec) maxConsec = cur
  }
  if (maxConsec >= 4) return false

  // 끝수(일의 자리) 중복 4개 이상 제외
  const lastDigitMap = new Map<number, number>()
  for (const n of sorted) {
    const d = n % 10
    lastDigitMap.set(d, (lastDigitMap.get(d) ?? 0) + 1)
  }
  if (Math.max(...lastDigitMap.values()) >= 4) return false

  // 같은 10번대에 5개 이상 제외
  const decadeMap = new Map<number, number>()
  for (const n of sorted) {
    const decade = Math.floor((n - 1) / 10)
    decadeMap.set(decade, (decadeMap.get(decade) ?? 0) + 1)
  }
  if (Math.max(...decadeMap.values()) >= 5) return false

  return true
}

export function parseNumbers(numbersStr: string): number[] {
  return numbersStr.split(',').map(Number)
}
```

// src/lib/lotto-api.ts
import { LottoDraw } from '@/types/lotto'
import { LOTTO_API_BASE } from './constants'
import { createComboKey } from './combo'

type LottoApiResponse = {
  returnValue: 'success' | 'fail'
  drwNo: number
  drwNoDate: string
  drwtNo1: number
  drwtNo2: number
  drwtNo3: number
  drwtNo4: number
  drwtNo5: number
  drwtNo6: number
  bnusNo: number
}

export async function fetchLottoDraw(round: number): Promise<LottoDraw | null> {
  try {
    const url = `${LOTTO_API_BASE}?method=getLottoNumber&drwNo=${round}`
    const res = await fetch(url, { cache: 'no-store' })
    if (!res.ok) return null

    const data: LottoApiResponse = await res.json()
    if (data.returnValue !== 'success') return null

    const numbers = [
      data.drwtNo1, data.drwtNo2, data.drwtNo3,
      data.drwtNo4, data.drwtNo5, data.drwtNo6,
    ].sort((a, b) => a - b)

    return {
      round: data.drwNo,
      drawDate: data.drwNoDate,
      numbers,
      bonus: data.bnusNo,
      comboKey: createComboKey(numbers),
    }
  } catch {
    return null
  }
}

// src/lib/analyzer.ts
import { NumberStats } from '@/types/lotto'

export type DrawRow = {
  round: number
  n1: number; n2: number; n3: number
  n4: number; n5: number; n6: number
}

function getNumbers(d: DrawRow): number[] {
  return [d.n1, d.n2, d.n3, d.n4, d.n5, d.n6]
}

export function analyzeNumbers(draws: DrawRow[]): NumberStats[] {
  if (draws.length === 0) return Array.from({ length: 45 }, (_, i) => ({
    number: i + 1, totalCount: 0, recent30Count: 0, recent100Count: 0,
    lastSeenRound: null, lastSeenGap: 0, averageGap: 0,
    hotScore: 0, coldScore: 0, finalScore: 0,
  }))

  const sorted = [...draws].sort((a, b) => a.round - b.round)
  const latest = sorted[sorted.length - 1].round
  const recent30 = sorted.slice(-30)
  const recent100 = sorted.slice(-100)

  return Array.from({ length: 45 }, (_, i) => {
    const n = i + 1
    const totalCount = sorted.filter(d => getNumbers(d).includes(n)).length
    const recent30Count = recent30.filter(d => getNumbers(d).includes(n)).length
    const recent100Count = recent100.filter(d => getNumbers(d).includes(n)).length

    const lastSeenDraw = [...sorted].reverse().find(d => getNumbers(d).includes(n))
    const lastSeenRound = lastSeenDraw?.round ?? null
    const lastSeenGap = lastSeenRound !== null ? latest - lastSeenRound : latest

    const averageGap = totalCount > 0 ? sorted.length / totalCount : sorted.length

    const hotScore = Math.min(100,
      (recent30Count / Math.max(1, recent30.length / 7.5)) * 60 +
      (recent100Count / Math.max(1, recent100.length / 7.5)) * 40
    )
    const coldScore = Math.min(100, (lastSeenGap / Math.max(1, averageGap)) * 100)
    const finalScore = Math.min(100, Math.round(hotScore * 0.55 + coldScore * 0.45))

    return {
      number: n, totalCount, recent30Count, recent100Count,
      lastSeenRound, lastSeenGap, averageGap, hotScore, coldScore, finalScore,
    }
  })
}

export function calcPairScores(draws: DrawRow[]): Map<string, number> {
  const pairMap = new Map<string, number>()
  for (const draw of draws) {
    const nums = getNumbers(draw).sort((a, b) => a - b)
    for (let i = 0; i < nums.length; i++) {
      for (let j = i + 1; j < nums.length; j++) {
        const key = `${nums[i]}-${nums[j]}`
        pairMap.set(key, (pairMap.get(key) ?? 0) + 1)
      }
    }
  }
  return pairMap
}

// src/lib/scorer.ts
import { NumberStats } from '@/types/lotto'
import { SCORE_WEIGHTS, LOW_NUMBER_THRESHOLD } from './constants'

export function calcComboScore(
  numbers: number[],
  statsMap: Map<number, NumberStats>,
  pairMap: Map<string, number>,
): number {
  const sorted = [...numbers].sort((a, b) => a - b)

  // 번호별 평균 점수
  const avgFinalScore = sorted.reduce((sum, n) => sum + (statsMap.get(n)?.finalScore ?? 0), 0) / 6

  // 합계 분포 점수 (115~165가 과거 최빈 범위)
  const comboSum = sorted.reduce((a, b) => a + b, 0)
  const sumScore = comboSum >= 115 && comboSum <= 165 ? 100 : comboSum >= 90 && comboSum <= 190 ? 60 : 0

  // 홀짝 균형 점수
  const oddCount = sorted.filter(n => n % 2 === 1).length
  const balanceScore = Math.abs(oddCount - 3) === 0 ? 100 : Math.abs(oddCount - 3) === 1 ? 70 : 30

  // 고저 균형 점수
  const lowCount = sorted.filter(n => n <= LOW_NUMBER_THRESHOLD).length
  const lowHighScore = Math.abs(lowCount - 3) === 0 ? 100 : Math.abs(lowCount - 3) === 1 ? 70 : 30

  // 페어 점수 (과거 자주 나온 쌍)
  let pairScore = 0
  for (let i = 0; i < sorted.length; i++) {
    for (let j = i + 1; j < sorted.length; j++) {
      pairScore += pairMap.get(`${sorted[i]}-${sorted[j]}`) ?? 0
    }
  }
  const maxPairScore = 50
  const normalizedPairScore = Math.min(100, (pairScore / maxPairScore) * 100)

  const raw =
    avgFinalScore * (SCORE_WEIGHTS.totalFrequency + SCORE_WEIGHTS.recent30Trend + SCORE_WEIGHTS.recent100Trend + SCORE_WEIGHTS.coldGap) +
    normalizedPairScore * (SCORE_WEIGHTS.pairFrequency + SCORE_WEIGHTS.tripleFrequency) +
    sumScore * SCORE_WEIGHTS.sumDistribution +
    ((balanceScore + lowHighScore) / 2) * SCORE_WEIGHTS.balance

  return Math.min(100, Math.round(raw))
}

// src/lib/generator.ts
import { GeneratedCombo } from '@/types/lotto'
import { DrawRow, analyzeNumbers, calcPairScores } from './analyzer'
import { isValidCombo, createComboKey } from './combo'
import { calcComboScore } from './scorer'
import { COMBO_CANDIDATES, LOW_NUMBER_THRESHOLD } from './constants'

export function generateCombos(draws: DrawRow[], count: number): GeneratedCombo[] {
  const stats = analyzeNumbers(draws)
  const pairMap = calcPairScores(draws)
  const statsMap = new Map(stats.map(s => [s.number, s]))

  const pastComboSet = new Set(draws.map(d =>
    createComboKey([d.n1, d.n2, d.n3, d.n4, d.n5, d.n6])
  ))

  // 가중치: finalScore 기반
  const weights = stats.map(s => Math.pow((s.finalScore + 10) / 60, 1.4))
  const totalWeight = weights.reduce((a, b) => a + b, 0)

  function weightedPick(exclude: Set<number>): number {
    let r = Math.random() * weights.reduce((sum, w, i) => {
      return exclude.has(i + 1) ? sum : sum + w
    }, 0)
    for (let i = 0; i < 45; i++) {
      if (!exclude.has(i + 1)) {
        r -= weights[i]
        if (r <= 0) return i + 1
      }
    }
    // fallback: 첫 번째 비제외 번호
    for (let i = 0; i < 45; i++) if (!exclude.has(i + 1)) return i + 1
    return 1
  }

  const results: GeneratedCombo[] = []
  const usedKeys = new Set<string>()

  for (let attempt = 0; attempt < COMBO_CANDIDATES && results.length < count; attempt++) {
    const picked = new Set<number>()
    while (picked.size < 6) picked.add(weightedPick(picked))

    const numbers = [...picked].sort((a, b) => a - b)
    const key = numbers.join('-')

    if (usedKeys.has(key)) continue
    if (!isValidCombo(numbers, pastComboSet)) continue

    usedKeys.add(key)

    const sum = numbers.reduce((a, b) => a + b, 0)
    const oddCount = numbers.filter(n => n % 2 === 1).length
    const evenCount = 6 - oddCount
    const lowCount = numbers.filter(n => n <= LOW_NUMBER_THRESHOLD).length
    const highCount = 6 - lowCount

    let consecutivePairCount = 0
    for (let i = 1; i < numbers.length; i++) {
      if (numbers[i] === numbers[i - 1] + 1) consecutivePairCount++
    }

    const score = calcComboScore(numbers, statsMap, pairMap)

    const reason: string[] = ['과거 당첨 이력이 없는 조합']
    if (sum >= 115 && sum <= 165) reason.push(`합계 ${sum} — 과거 주요 분포 범위`)
    else reason.push(`합계 ${sum} — 유효 범위 내`)
    if (Math.abs(oddCount - 3) <= 1) reason.push(`홀짝 균형 ${oddCount}:${evenCount}`)
    if (Math.abs(lowCount - 3) <= 1) reason.push(`저고 균형 ${lowCount}:${highCount}`)
    if (consecutivePairCount > 0) reason.push(`연속번호 ${consecutivePairCount}쌍 포함`)

    results.push({ numbers, score, sum, oddCount, evenCount, lowCount, highCount, consecutivePairCount, reason })
  }

  return results.sort((a, b) => b.score - a.score).slice(0, count)
}

import { NextResponse } from 'next/server'
import { prisma } from '@/lib/db'
import { DistributionStats } from '@/types/lotto'
import { LOW_NUMBER_THRESHOLD } from '@/lib/constants'

export async function GET() {
  try {
    const draws = await prisma.draw.findMany({
      orderBy: { round: 'asc' },
      select: { n1: true, n2: true, n3: true, n4: true, n5: true, n6: true },
    })

    // 합계 분포 (10 단위 버킷)
    const sumBuckets = new Map<string, number>()
    const oddEvenMap = new Map<string, number>()
    const lowHighMap = new Map<string, number>()
    const consecMap = new Map<number, number>()

    for (const d of draws) {
      const nums = [d.n1, d.n2, d.n3, d.n4, d.n5, d.n6].sort((a, b) => a - b)
      const sum = nums.reduce((a, b) => a + b, 0)
      const bucket = `${Math.floor(sum / 10) * 10}~${Math.floor(sum / 10) * 10 + 9}`
      sumBuckets.set(bucket, (sumBuckets.get(bucket) ?? 0) + 1)

      const oddCount = nums.filter(n => n % 2 === 1).length
      const oddKey = `${oddCount}홀${6 - oddCount}짝`
      oddEvenMap.set(oddKey, (oddEvenMap.get(oddKey) ?? 0) + 1)

      const lowCount = nums.filter(n => n <= LOW_NUMBER_THRESHOLD).length
      const lowKey = `저${lowCount}고${6 - lowCount}`
      lowHighMap.set(lowKey, (lowHighMap.get(lowKey) ?? 0) + 1)

      let pairs = 0
      for (let i = 1; i < nums.length; i++) {
        if (nums[i] === nums[i - 1] + 1) pairs++
      }
      consecMap.set(pairs, (consecMap.get(pairs) ?? 0) + 1)
    }

    const result: DistributionStats = {
      sumDistribution: [...sumBuckets.entries()]
        .map(([range, count]) => ({ range, count }))
        .sort((a, b) => parseInt(a.range) - parseInt(b.range)),
      oddEvenDistribution: [...oddEvenMap.entries()].map(([label, count]) => ({ label, count })),
      lowHighDistribution: [...lowHighMap.entries()].map(([label, count]) => ({ label, count })),
      consecutiveFrequency: [...consecMap.entries()]
        .map(([pairs, frequency]) => ({ pairs, frequency }))
        .sort((a, b) => a.pairs - b.pairs),
    }

    return NextResponse.json(result)
  } catch {
    return NextResponse.json({ error: '분포 통계 조회 실패' }, { status: 500 })
  }
}

import { prisma } from '@/lib/db'
import { createComboKey, parseNumbers } from '@/lib/combo'
import { MAX_SAVED_COMBOS } from '@/lib/constants'

export async function GET() {
  try {
    const saved = await prisma.savedCombo.findMany({ orderBy: { createdAt: 'desc' } })
    const result = saved.map(s => ({
      id: s.id,
      numbers: parseNumbers(s.numbers),
      comboKey: s.comboKey,
      score: s.score,
      matchedRound: s.matchedRound,
      createdAt: s.createdAt.toISOString(),
    }))
    return NextResponse.json({ savedCombos: result })
  } catch {
    return NextResponse.json({ error: '조회 실패' }, { status: 500 })
  }
}

export async function POST(req: NextRequest) {
  try {
    const body = await req.json()
    const numbers: number[] = body.numbers
    const score: number | undefined = body.score

    if (!Array.isArray(numbers) || numbers.length !== 6) {
      return NextResponse.json({ error: '번호는 6개여야 합니다' }, { status: 400 })
    }
    if (numbers.some(n => !Number.isInteger(n) || n < 1 || n > 45)) {
      return NextResponse.json({ error: '번호는 1~45 사이 정수여야 합니다' }, { status: 400 })
    }
    if (new Set(numbers).size !== 6) {
      return NextResponse.json({ error: '중복 번호가 있습니다' }, { status: 400 })
    }

    const count = await prisma.savedCombo.count()
    if (count >= MAX_SAVED_COMBOS) {
      return NextResponse.json({ error: `저장 가능한 조합은 최대 ${MAX_SAVED_COMBOS}개입니다. 기존 조합을 삭제한 뒤 다시 저장해주세요.` }, { status: 400 })
    }

    const sorted = [...numbers].sort((a, b) => a - b)
    const comboKey = createComboKey(sorted)
    const numbersStr = sorted.join(',')

    // 이미 당첨 이력이 있는 조합인지 확인
    const matchedDraw = await prisma.draw.findUnique({ where: { comboKey } })

    const saved = await prisma.savedCombo.upsert({
      where: { comboKey },
      update: {},
      create: {
        numbers: numbersStr,
        comboKey,
        score: score ?? null,
        matchedRound: matchedDraw?.round ?? null,
        matchedDrawId: matchedDraw?.id ?? null,
      },
    })

    return NextResponse.json({
      savedCombo: {
        id: saved.id,
        numbers: sorted,
        comboKey: saved.comboKey,
        score: saved.score,
        matchedRound: saved.matchedRound,
        createdAt: saved.createdAt.toISOString(),
      }
    }, { status: 201 })
  } catch (e: any) {
    if (e?.code === 'P2002') {
      return NextResponse.json({ error: '이미 저장한 조합입니다' }, { status: 409 })
    }
    return NextResponse.json({ error: '저장 실패' }, { status: 500 })
  }
}

          if [ "$response" != "200" ]; then
            echo "Sync failed with status $response"
            exit 1
          fi
```
